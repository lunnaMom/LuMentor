            }
        }
        
        function checkQuiz(quizNum) {
            const q1 = document.querySelector('input[name="q1"]:checked');
            const q2 = document.querySelector('input[name="q2"]:checked');
            const result = document.getElementById(`quizResult${quizNum}`);
            
            let score = 0;
            if (q1 && q1.value === 'b') score++;
            if (q2 && q2.value === 'c') score++;
            
            if (score === 2) {
                result.innerHTML = '<div class="result success">üéâ –ò–¥–µ–∞–ª—å–Ω–æ! –í—Å–µ –æ—Ç–≤–µ—Ç—ã –ø—Ä–∞–≤–∏–ª—å–Ω—ã–µ! +20 –æ—á–∫–æ–≤</div>';
                totalPoints += 20;
                completedLessons++;
                unlockNextLesson();
            } else {
                result.innerHTML = `<div class="result error">–ü—Ä–∞–≤–∏–ª—å–Ω–æ ${score} –∏–∑ 2. –ü–æ–ø—Ä–æ–±—É–π –µ—â—ë —Ä–∞–∑!</div>`;
            }
            updateStats();
        }
        
        function unlockNextLesson() {
            document.getElementById('nextBtn').style.display = 'block';
            const nextLesson = document.getElementById(`lesson${currentLesson + 1}`);
            if (nextLesson) {
                nextLesson.querySelector('.lesson-status').textContent = 'üìù';
                nextLesson.classList.remove('locked');
            }
        }
        
        function nextLesson() {
            currentLesson++;
            // –ó–¥–µ—Å—å –±—É–¥–µ—Ç –∫–æ–¥ –¥–ª—è –æ—Ç–∫—Ä—ã—Ç–∏—è —Å–ª–µ–¥—É—é—â–µ–≥–æ —É—Ä–æ–∫–∞
            alert('–û—Ç–ª–∏—á–Ω–æ! –°–∫–∞–∂–∏ "–î–∞–ª—å—à–µ" —á—Ç–æ–±—ã —è –¥–æ–±–∞–≤–∏–ª —Å–ª–µ–¥—É—é—â–∏–π —É—Ä–æ–∫!');
        }
        
        function updateStats() {
            document.getElementById('completedLessons').textContent = completedLessons;
            document.getElementById('totalPoints').textContent = totalPoints;
            document.getElementById('currentLevel').textContent = 8 - completedLessons;
            
            const progressPercent = (completedLessons / 8) * 100;
            document.getElementById('progressFill').style.width = progressPercent + '%';
        }
